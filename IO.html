#导入模块 package IO; use strict; use warnings; use Tk; use Tk::MsgBox; use utf8; use Fcntl qw(:flock); require Exporter; our @ISA = qw(Exporter);

#设置程式名称 my $appname = 'IO';

#设置版本号 my $version = 1.0;

#new方法dc sub new {

my $caller = shift; my \(class = ref(\)caller) || $caller;

my $self = {};

bless \(self, \)class; return $self;

}

#名字 :ReadFile #功能 :读取文本文件 #参数 :文件路径 #返回值 :文件内容 #使用例子 :$h->ReadFile('c:/test.txt'); sub ReadFile{

my $self = shift; my $filePath = shift; my @file;

eval{ open FILE,\(filePath or die "can't open \)filePath $!\n"; @file = <FILE>; close FILE; #unlink $filePath;

};

\(self->StopMsgBox('error',"读取文件失败！\n \)@") if $@;

return @file;

}

#名字 :WriteFile #功能 :写文本文件 #参数 :1,文件路径 2,文件内容 #返回值 :1 #使用例子 :$h->WriteFile('c:/test.txt','test'); sub WriteFile{

my $self = shift; my $filePath = shift; my $file = shift; my $model = shift || '>>';

#写入文本文件

eval{

open FILE,\(model,\)filePath or die "can't open \(filePath \)!\n"; flock (FILE,LOCK_EX); print FILE "$file\n"; flock (FILE,LOCK_UN); close FILE;

};
\(self->StopMsgBox('error',"写入文件失败！\n \)@") if $@; return 1;

}

#名字 :SetUser #功能 :设置显示图片 #参数 :设置显示内容 #返回值 :无 #使用例子 :$h->StopMsgBox('error',"工作层是顶层,添加text字无需镜像"); sub StopMsgBox{

my $self = shift; my $icon = shift; my $msg = shift; my $mw = MainWindow -> new;

$mw->withdraw;

$mw->optionAdd("*font" => 'application 11');
my \(d = \)mw->MsgBox('-icon' => $icon, -type => 'OK', -title => '提示信息', -message => $msg,); $d-> transient(''); $d->Show; $mw ->destroy;

}

1;
