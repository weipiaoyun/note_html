获取设备信息

"{ "Reason": "OK", "ResultCode": "1", "Result": [{ "Key": "设备0", "Value": 0 }, { "Key": "设备1", "Value": 0 } ]}"

"{"Reason": "OK", "ResultCode": "1", "Result": { "Name": "内层棕化", "Status": 0 }}"
