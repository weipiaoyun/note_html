Public Function IsValidataPwd(str_UserNo As String, str_UserPwd As String) As String Dim str_Sql As String = "Update T_sys_User set LastLoginTime=@LastLoginTime where UserNo=@UserNo and UserPwd=@UserPwd" Dim a As Int32 = Help_DataBase.ExecuteNonQuery(str_Sql, Factory, CommandType.Text, Help_DataBase.SqlParameter("@UserNo", str_UserNo), Help_DataBase.SqlParameter("@UserPwd", str_UserPwd), Help_DataBase.SqlParameter("@LastLoginTime", DateTime.Now)) If a >= 1 Then Return Nothing Else Return "密码错误" End If End Function
