#!/usr/local/bin/perl -w

use strict;

use FileHandle;

use utf8;

use 5.010;

use storable qw(store retrieve); # 表明用到了模块中的两个函数。

  1. store 用途: 在一个二进制文件中保存数据结构
  1. retrieve 用途: 从一个二进制文件中读出数据结构
  1. 数组结构

my $myArray = ['a','b','b','d','f'];

  1. 散列结构

my $myHash = {a => 1,b => 2,c => 3,d => 4,e => 5,};

  1. 保存数组结构

my \(statusArray = store \)myArray, "TestArray.dat"; # 如果成功返回true，不成功返回未定义。

  1. 保存数组结构

my \(statusHash = store \)myHash, "TestHash.dat"; # 如果成功返回true，不成功返回未定义。

  1. 读取数组结构

my $testArray = retrieve("TestArray.dat");

  1. 读取散列结构

my $testHash = retrieve("TestHash.dat");

say 'array=>',Dumper($testArray);

say 'hash=>',Dumper($testHash);
