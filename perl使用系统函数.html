$ViewFilePath = "/genesis/sys/scripts/";
@filename = ();
opendir Dir,"$ViewFilePath";
foreach my $file (readdir Dir) {
if ( ! -d $file ) {
push @filename , $file;
}
}
@filename = reverse(@filename);
