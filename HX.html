流程参数

update Data0494 SET PARAMETER_VALUE = '0.13' WHERE rkey = ( select d494.RKEY

FROM Data0050 as d50 inner join Data0025 as d25 on d50.RKEY = d25.ANCESTOR_PTR

INNER JOIN dbo.Data0038 AS D38 ON d25.RKEY = D38.SOURCE_PTR

INNER JOIN dbo.Data0494 as d494 on d494.ROUTE_STEP_PTR = D38.RKEY INNER JOIN dbo.Data0278 AS D278 ON D494.PARAMETER_PTR = D278.RKEY INNER JOIN dbo.Data0034 AS D34 ON D34.RKEY = d38.DEPT_PTR

where d50.CUSTOMER_PART_NUMBER + d50.CP_REV = 's08i001815a0' and d278.RKEY = '2222'

select d25.MANU_PART_NUMBER ,d494.PARAMETER_VALUE,d278.PARAMETER_DESC,d278.RKEY ,d38.STEP_NUMBER,d34.DEPT_NAME,d38.FLOW_NO

FROM Data0050 as d50 inner join Data0025 as d25 on d50.RKEY = d25.ANCESTOR_PTR

INNER JOIN dbo.Data0038 AS D38 ON d25.RKEY = D38.SOURCE_PTR

INNER JOIN dbo.Data0494 as d494 on d494.ROUTE_STEP_PTR = D38.RKEY INNER JOIN dbo.Data0278 AS D278 ON D494.PARAMETER_PTR = D278.RKEY INNER JOIN dbo.Data0034 AS D34 ON D34.RKEY = d38.DEPT_PTR where d50.CUSTOMER_PART_NUMBER like '%s08i001815%' order by d38.STEP_NUMBER
