perl mysql

use DBI;

#设置mysql连接参数 my $sqlDataBase = 'test'; #设置数据库名称 my $sqlHost = 'localhost'; #设置主机名称 my $sqlUser = 'root'; #设置用户 my $sqlPwd = 'lwtest';#设置密码 my \(dsn = "DBI:mysql:database=\)sqlDataBase; host = $sqlHost";#设置dsn(包括主机名称和数据库名称，数据库类型) my $attributes = {'RaiseError' => 1}; #设置其他

#连接mysql my \(dbh = DBI->connect(\)dsn, \(sqlUser, \)sqlPwd, $attributes); #dsn 用户，密码，其他

#插入 my \(rows = \)dbh->do("INSERT INTO users VALUES (4, 'jay', 'CZ'), (5, 'll', 'KJ')");

#断开连接 $dbh->disconnect(); #断开连接

linux创建mysql新用户并添加权限

  1. mysql -uroot -p #登录
  2. use iasystem #切换到库iasystem
  3. insert into mysql.user(Host,User,Passwd)
  4. update mysql.user set Create_priv = 'Y' where user = 'admin'
  5. grant all privileges on . to 'admin'@'%' identified by '123456';
  6. select * from mysql.user where user = 'admin';
  7. /etc/init.d/mysqld restart #重启mysql服务器
