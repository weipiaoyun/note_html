/****** Script for SelectTopNRows command from SSMS ******/
SELECT TOP 1000 [rd_id]

,[cp_id] ,[location] ,[device_id] ,[cp_name] ,[cp_ip] ,[canteen_pos_enabled] ,[ufdid] ,[user_location] ,[uid] ,[user_name] ,[purchase_time] ,[purchase_date] ,[period_id] ,[period_name] ,[year_week] ,[year_month] ,[purchase_money] ,[subsidize_money] ,[actual_money] ,[delete] ,[insert_time]

FROM [iqps].[ism].[v_canteen_pos_record_detail] WHERE user_name = '林伟' ORDER BY purchase_time DESC
