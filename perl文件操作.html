拖文件夹到perl

#!c:/perl/bin/perl -w
use warnings;
use Encode;
use Win32::PerfLib;
use Win32::GUI();
use Win32::GUI::DropFiles();

my $mw = Win32::GUI::Window->new(
-title => "Win32::GUI拖放",
-pos => [100,100],
-size => [400,300],
-onResize => \&mwResize,
);

$mw->AddLabel(
-pos => [10,10],
-text => "把文件拖人列表中",
);

$mw->AddListbox(
-name => 'list1',
-pos => [10,30],
-vscroll => 1,
-acceptfiles => 1,
-onDropFiles => \&gotDrop,
);

$mw->Show();
Win32::GUI::Dialog();
$mw->Hide();
exit(0);

sub gotDrop {
my (\(self, \)dropObj) = @_;
my @rr = $dropObj->GetDroppedFiles();
my \(path = Encode::encode('gbk',\)rr[0]);
Win32::GUI::MessageBox(0,"$path","提示",0x0003|0x0030);
}

sub mwResize {
my $self = shift;

$self->list1->Resize(
$self->ScaleWidth()-20,
$self->ScaleHeight()-40,
);

return 1;
}

TK拖动文件

#!/usr/bin/perl -w
use Tk;
use Tk::DropSite;

$mw = MainWindow->new();
my \(txt = \)mw->Text()->pack();
my \(drop = \)txt->DropSite(-droptypes,'Win32', -dropcommand, [\&dropCmd, $txt]);

MainLoop;

sub dropCmd {
my (\(txt, \)s) = @_;
my \(filename = \)txt->SelectionGet(-selection => $s,'STRING');
\(txt->insert('end', \)filename."\n");
}
